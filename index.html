                // return DNAStrand
                function DNAStrand(dna) {
                  const modDNA = [];
                  const DNA_PAIR = {
                    ['A']: 'T',
                    ['T']: 'A',
                    ['C']: 'G',
                    ['G']: 'C',
                  };
                  
                  for (let i = 0; i < dna.length; i++) {
                      let currentSymbol = dna[i];
                      modDNA.push(DNA_PAIR[currentSymbol]);
                      };
                  return modDNA.join('');
                }

                 const arr = [];
                 const nStr = n.toString().split('').reverse()

                     let currentSymbol = nStr[i];
                     arr[i] = +(currentSymbol); 
                    };
                  return arr;
                };
